def float_to_q1_10_10(val):
    scale = 2 ** 10
    raw = int(round(val * scale))  # 乘上小數精度
    if raw < 0:
        raw = (1 << 21) + raw       # 轉二補數
    return format(raw, '024b'), hex(raw)

# 使用範例
print(float_to_q1_10_10(25.625))  # ('000000000110011010000', '0x3340')
print(float_to_q1_10_10(-3.125))  # ('111111111111001110000', '0xffff9c0')
